73 public HashMap<Item, Double> recommended_items() {

84 //If item already in the map take the greatest rating prediction
85 else if (predictions.containsKey(key)) {
86 if (predictions.get(key) < knn.get(key))
87 predictions.replace(key, knn.get(key));
88 }
89 else if (predictions.size() < k) {
90 if (predictions.get(minimum) > knn.get(key))
91 minimum = key;
92 predictions.put(key, knn.get(key));
93 }
94 else if (predictions.get(minimum) < knn.get(key)) {
95 predictions.remove(minimum);
96 predictions.put(key, knn.get(key));
97 minimum = Collections.min(predictions.entrySet(), Map.Entry.comparingByValue()).getKey();
98 }
99 }

114 private HashMap<Item, Double> get_KNN(Item a, double rating) {
115 SortedMap<Double, List<Item>> maxPred = new TreeMap<>();
116 HashMap<String,Item> data = super.dataset.getItems();
117 for (Item dItem : data.values()) {
118 if (!ratedItems.containsKey(dItem)) {
119 DistanceItems d = new DistanceItems(a, dItem, dataset.getMaxValIntegers().values(), dataset.getMinValIntegers().values(),
120 dataset.getMaxValDoubles().values(), dataset.getMinValDoubles().values(), dataset.getMaxDates().values(),
121 dataset.getMinDates().values(), intAt, doubAt, dateAt, bAt, catAt, ftAt);
122 double prediction = (1 - d.get_distance()) * maxRating - (maxRating - rating) / (2*maxRating);
123 if (super.ratings.containsKey((dItem))) {
124 double meanRating = 0;
125 ArrayList<Double> stars = super.ratings.get(dItem);
126 for (double rate : stars)
127 meanRating += rate;
128 meanRating /= stars.size();
129 prediction -= (maxRating - meanRating) / (2*maxRating);
130
131 }
132 if (prediction < 0) prediction = 0.0;
133 if(maxPred.containsKey(prediction)) {
134 maxPred.get(prediction).add(dItem);
135 }
136 else {
137 List<Item> value = new ArrayList<>(List.of(dItem));
138 if (maxPred.size() < k)
139 maxPred.put(prediction, value);
140 else if (maxPred.firstKey() < prediction) {
141 maxPred.remove(maxPred.firstKey());
142 maxPred.put(prediction, value);
143 }
144 }
145 }
146 }
147 HashMap<Item, Double> knn = new HashMap<>();
148 int itemCounter = 0;
149 for (Double d : maxPred.keySet()) {
150 for (int i = 0; i < maxPred.get(d).size() && itemCounter < k; ++i, ++itemCounter) {
151 knn.put(maxPred.get(d).get(i), d);
152 }
153 if (itemCounter == k) break;
154 }
155 return knn;
